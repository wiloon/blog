查看字符集设置

[shell]
   
show variables like 'character\_set\_%';
  
[/shell]
  
[shell]
  
set character\_set\_client=utf8;
  
set character\_set\_connection=utf8;
  
set character\_set\_database=utf8;
  
set character\_set\_results=utf8;
  
set character\_set\_server=utf8;
  
[/shell]

set character\_set\_database=utf8;
  
..character\_set\_client
  
..character\_set\_connection
  
..character\_set\_database

sudo emacs /etc/mysql/my.cnf
  
在[client]下添加
  
default-character-set=utf8

在[mysqld]下添加
  
default-character-set=utf8 (!!!mysql 5.5.20 以后的版本要这样改 character-set-server=utf8)

[code lang=sql]
  
- 查看表编码
  
SHOW CREATE TABLE tbl_name;
  
```

改表的字符集.
  
ALTER TABLE tbl_name CONVERT TO CHARACTER SET utf8;

重启mysql;
  
sudo /etc/init.d/mysql restart;
