;insert line
  
(global-set-key (kbd "S-<return>") '(lambda()(interactive)(move-end-of-line 1)(newline)))
  
(global-set-key (kbd "C-S-<return>") '(lambda()(interactive)(move-beginning-of-line 1)(newline)(previous-line)))
