;insert line
  
(global-set-key (kbd &#8220;S-<return>&#8221;) &#8216;(lambda()(interactive)(move-end-of-line 1)(newline)))
  
(global-set-key (kbd &#8220;C-S-<return>&#8221;) &#8216;(lambda()(interactive)(move-beginning-of-line 1)(newline)(previous-line)))
