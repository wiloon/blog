C-c m d

[shell]

(defun my-insert-date ()

(interactive)
   
(insert "#")
   
;(insert (user-full-name))
   
;(insert "@")
   
(insert (format-time-string "%Y/%m/%d %H:%M:%S" (current-time))))
   
(global-set-key (kbd "C-c m d") &#8216;my-insert-date)

[/shell]
