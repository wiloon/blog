var fwidth = $(&#8220;#footer&#8221;).height();

var scrollTop = $(window).scrollTop();
  
var scrollBtm = $(document).height() &#8211; $(window).scrollTop() &#8211; $(&#8220;#widget&#8221;).height();

}
  
} else {

}
  
})
