build executable file

create systemd unit config file /usr/lib/systemd/system/foo

add following lines

[Unit]
  
Description=foo

[Service]
  
User=root
  
Group=root
  
Restart=on-failure
  
ExecStart=/path/to/execfile/foo

[Install]
  
WantedBy=multi-user.target
